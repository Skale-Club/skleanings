function shiftHHMM(hhmm: string, minutes: number): string {
  const [h, m] = hhmm.split(':').map(Number);
  const total = h * 60 + m + minutes;
  const newH = Math.floor(Math.max(0, total) / 60) % 24;
  const newM = Math.max(0, total) % 60;
