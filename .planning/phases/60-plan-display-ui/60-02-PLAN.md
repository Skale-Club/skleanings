```typescript
// =============================================================================
// useUpdateTenantPlan — PATCH /api/super-admin/tenants/:id/plan (Phase 60, PT-07)
// =============================================================================

export function useUpdateTenantPlan() {
  const queryClient = useQueryClient();
  return useMutation<
    { message: string; planTier: string; subscription: unknown },
    Error,
    { tenantId: number; planTier: string }
  >({
    mutationFn: ({ tenantId, planTier }) =>
      superAdminFetch(`/api/super-admin/tenants/${tenantId}/plan`, {
        method: "PATCH",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ planTier }),
      }),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ["/api/super-admin/tenants"] });
    },
  });
}
```

Rationale (per spec note in execution_context): use `superAdminFetch` instead of a hand-rolled `fetch` so error handling (status, message extraction) matches the rest of the file. The hook returns the server's `{ message, planTier, subscription }` payload — surface any error via `.error` on the consumer side.
