```tsx
<TableCell>
  {(() => {
    const bs = tenant.billingStatus;
    if (!bs || bs === "none") {
      return <Badge className="bg-gray-100 text-gray-600">—</Badge>;
    }
    const cls =
      bs === "active"
        ? "bg-green-100 text-green-800"
        : bs === "past_due"
          ? "bg-amber-100 text-amber-800"
          : bs === "canceled"
            ? "bg-red-100 text-red-800"
            : "bg-gray-100 text-gray-600";
    return (
      <div className="space-y-0.5">
        <Badge className={cls}>{bs}</Badge>
        {tenant.billingPlanId && (
          <div className="text-xs text-gray-500 font-mono">{tenant.billingPlanId}</div>
        )}
        {tenant.billingPeriodEnd && (
          <div className="text-xs text-gray-400">
            {new Date(tenant.billingPeriodEnd).toLocaleDateString()}
          </div>
        )}
      </div>
    );
  })()}
</TableCell>
```

Place the IIFE pattern to keep the conditional rendering inline without extracting a helper function — consistent with the existing badge pattern in the Status column.
