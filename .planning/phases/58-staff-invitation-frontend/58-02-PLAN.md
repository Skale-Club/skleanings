     queryKey: ['/api/admin/staff/invitations'],
     queryFn: async () => {
       const res = await fetch('/api/admin/staff/invitations', { credentials: 'include' });

5. **Invite mutation**:
   ```ts
   const inviteMutation = useMutation({
     mutationFn: async (body: { email: string; role: 'staff' | 'admin' }) => {
       const res = await fetch('/api/admin/staff/invite', {
         method: 'POST',
         credentials: 'include',
         headers: { 'Content-Type': 'application/json' },
         body: JSON.stringify(body),
       });
       if (!res.ok) {
         const err = await res.json().catch(() => ({ message: 'Failed to send invitation' }));
         throw new Error(err.message ?? 'Failed to send invitation');
       }
       return res.json();
     },
     onSuccess: () => {
       queryClient.invalidateQueries({ queryKey: ['/api/admin/staff/invitations'] });
       toast({ title: 'Invitation sent', description: `An invite was sent to ${emailInput}.` });
       setIsInviteOpen(false);
       setEmailInput('');
       setRoleInput('staff');
       setInviteError(null);
     },
     onError: (err: Error) => {
       setInviteError(err.message);
     },
   });
   ```

6. **Revoke mutation**:
   ```ts
   const revokeMutation = useMutation({
     mutationFn: async (id: number) => {
       const res = await fetch(`/api/admin/staff/invite/${id}`, {
         method: 'DELETE',
         credentials: 'include',
       });
       if (!res.ok) {
         const err = await res.json().catch(() => ({ message: 'Failed to revoke' }));
         throw new Error(err.message ?? 'Failed to revoke');
       }
       return res.json();
     },
     onSuccess: () => {
       queryClient.invalidateQueries({ queryKey: ['/api/admin/staff/invitations'] });
       toast({ title: 'Invitation revoked' });
       setInvitationToRevoke(null);
     },
     onError: (err: Error) => {
       toast({ title: 'Error', description: err.message, variant: 'destructive' });
       setInvitationToRevoke(null);
     },
   });
   ```
