4. Add a useEffect to fetch invoices after the existing billing status useEffect:
```typescript
useEffect(() => {
  if (!isAuthenticated) return;
  fetch('/api/billing/invoices', { credentials: 'include' })
    .then(async (res) => {
      if (!res.ok) throw new Error('Failed to load invoices');
      return res.json();
    })
    .then((data: { invoices: Invoice[] }) => {
      setInvoices(data.invoices);
      setInvoicesLoading(false);
    })
    .catch(() => {
      setInvoicesLoading(false);
    });
}, [isAuthenticated]);
```

5. In the JSX return, add the Invoice History Card immediately after the closing tag of the existing `<Card>` (Subscription Status card):
```tsx
<Card className="mt-4">
  <CardHeader>
    <CardTitle>Invoice History</CardTitle>
  </CardHeader>
  <CardContent>
    {invoicesLoading ? (
      <div className="space-y-2">
        {[...Array(3)].map((_, i) => (
          <Skeleton key={i} className="h-10 w-full" />
        ))}
      </div>
    ) : invoices.length === 0 ? (
      <p className="text-sm text-muted-foreground">No invoices yet.</p>
    ) : (
      <Table>
        <TableHeader>
          <TableRow>
            <TableHead>Date</TableHead>
            <TableHead>Amount</TableHead>
            <TableHead>Status</TableHead>
            <TableHead>Download</TableHead>
          </TableRow>
        </TableHeader>
        <TableBody>
          {invoices.map((inv) => (
            <TableRow key={inv.id}>
              <TableCell>{new Date(inv.date).toLocaleDateString()}</TableCell>
              <TableCell>
                {(inv.amount / 100).toFixed(2)} {inv.currency.toUpperCase()}
              </TableCell>
              <TableCell>
                <Badge className={inv.status === 'paid' ? 'bg-green-100 text-green-800' : 'bg-gray-100 text-gray-600'}>
                  {inv.status}
                </Badge>
              </TableCell>
              <TableCell>
                {inv.invoiceUrl ? (

                    href={inv.invoiceUrl}

                  </a>
                ) : (
                  '—'
                )}

            </TableRow>
          ))}
        </TableBody>
      </Table>
    )}
  </CardContent>
</Card>
```
