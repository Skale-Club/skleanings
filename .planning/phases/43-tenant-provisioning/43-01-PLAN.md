      email: string,

    ): Promise<{ userId: string }> {
      return await db.transaction(async (tx) => {
        const userId = crypto.randomUUID();
        await tx.insert(users).values({

          tenantId,
          email,
          password: hashedPassword,
          role: "admin",

          tenantId,
          role: "admin",
        });
        return { userId };
      });

      tenantId: number,

      await db

        .values({
          tenantId,
          companyName,
          timeZone: "America/New_York",
          language: "en",
        })
        .onConflictDoNothing();
