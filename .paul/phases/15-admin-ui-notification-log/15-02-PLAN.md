<objective>
## Goal
Add a "Notification Log" section to the admin sidebar with a global paginated notification log table, filter bar, and preview modal — fetched from GET /api/admin/notification-logs.

## Purpose
Admins can browse all notification events across all conversations and bookings in one place, filter by channel/status/date/recipient, and read full message previews without leaving the admin UI.

## Output
- types.ts: `'notification-logs'` added to AdminSection union
- NotificationLogsSection.tsx: new admin section with filter bar, table, pagination, and preview modal
- Admin.tsx: Bell icon import, NotificationLogsSection import, menu item, render condition
</objective>

```gherkin
When the Notification Log section loads
Then a table renders with columns: Channel (icon), Recipient, Trigger, Status badge, Timestamp
```

## AC-3: Filter Bar Works
```gherkin
Given the Notification Log section is active
When the admin selects a channel filter (e.g. "sms")
Then the table refetches showing only SMS rows

```

```gherkin

Then a Dialog opens showing the full preview text
```

```gherkin

When the table renders
Then a "No notifications found" empty state is shown
```

```gherkin
Given more than 50 notification log rows exist
When the table reaches the last row
Then a "Next" button is available to load the next page
And a "Previous" button is available when not on the first page
```

<task type="auto">
  <name>Task 2: Create NotificationLogsSection.tsx</name>
  <files>client/src/components/admin/NotificationLogsSection.tsx</files>
  <action>
    Create a new file with the full component:

    ```tsx
    import { useState } from 'react';
    import { useQuery } from '@tanstack/react-query';
    import { authenticatedRequest } from '@/lib/queryClient';
    import { Phone, MessageCircle, Link2, Bell, ChevronLeft, ChevronRight, Loader2 } from 'lucide-react';
    import {
      Table,
      TableBody,
      TableCell,
      TableHead,
      TableHeader,
      TableRow,
    } from '@/components/ui/table';
    import {
      Select,
      SelectContent,
      SelectItem,
      SelectTrigger,
      SelectValue,
    } from '@/components/ui/select';
    import { Input } from '@/components/ui/input';
    import { Button } from '@/components/ui/button';
    import {
      Dialog,
      DialogContent,
      DialogHeader,
      DialogTitle,
    } from '@/components/ui/dialog';

    interface NotificationLog {
      id: number;
      channel: string;
      trigger: string;
      recipient: string;
      preview: string;
      status: string;
      errorMessage?: string | null;
      providerMessageId?: string | null;
      sentAt: string;
      conversationId?: string | null;
      bookingId?: number | null;
    }

    const LIMIT = 50;

    function channelIcon(channel: string) {
      if (channel === 'sms') return <Phone className="h-3.5 w-3.5" />;
      if (channel === 'telegram') return <MessageCircle className="h-3.5 w-3.5" />;
      return <Link2 className="h-3.5 w-3.5" />;
    }

    function statusClass(status: string) {
      if (status === 'sent') return 'bg-green-100 text-green-700 dark:bg-green-900/30 dark:text-green-400';
      if (status === 'failed') return 'bg-red-100 text-red-700 dark:bg-red-900/30 dark:text-red-400';
      return 'bg-muted text-muted-foreground';
    }

    export function NotificationLogsSection({
      getAccessToken,
    }: {
      getAccessToken: () => Promise<string | null>;
    }) {
      const [channel, setChannel] = useState('');
      const [status, setStatus] = useState('');
      const [search, setSearch] = useState('');
      const [from, setFrom] = useState('');
      const [to, setTo] = useState('');
      const [offset, setOffset] = useState(0);
      const [selectedLog, setSelectedLog] = useState<NotificationLog | null>(null);

      const { data: logs = [], isLoading } = useQuery<NotificationLog[]>({
        queryKey: ['/api/admin/notification-logs', { channel, status, search, from, to, offset }],
        queryFn: async () => {
          const token = await getAccessToken();
          if (!token) return [];
          const params = new URLSearchParams();
          if (channel) params.set('channel', channel);
          if (status) params.set('status', status);
          if (search) params.set('search', search);
          if (from) params.set('from', from);
          if (to) params.set('to', to);
          params.set('limit', String(LIMIT));
          params.set('offset', String(offset));
          const res = await authenticatedRequest('GET', `/api/admin/notification-logs?${params}`, token);
          if (!res.ok) throw new Error('Failed to fetch notification logs');
          return res.json();
        },
        staleTime: 30_000,
      });

      const handleReset = () => {
        setChannel('');
        setStatus('');
        setSearch('');
        setFrom('');
        setTo('');
        setOffset(0);
      };

      return (
        <div className="space-y-4">
          <h2 className="text-xl font-semibold">Notification Log</h2>

          {/* Filter bar */}
          <div className="flex flex-wrap gap-2 items-end">
            <Select value={channel || 'all'} onValueChange={(v) => { setChannel(v === 'all' ? '' : v); setOffset(0); }}>
              <SelectTrigger className="w-36 h-8 text-sm">
                <SelectValue placeholder="Channel" />
              </SelectTrigger>
              <SelectContent>
                <SelectItem value="all">All channels</SelectItem>
                <SelectItem value="sms">SMS</SelectItem>
                <SelectItem value="telegram">Telegram</SelectItem>
                <SelectItem value="ghl">GHL</SelectItem>
              </SelectContent>
            </Select>

            <Select value={status || 'all'} onValueChange={(v) => { setStatus(v === 'all' ? '' : v); setOffset(0); }}>
              <SelectTrigger className="w-32 h-8 text-sm">
                <SelectValue placeholder="Status" />
              </SelectTrigger>
              <SelectContent>
                <SelectItem value="all">All statuses</SelectItem>
                <SelectItem value="sent">Sent</SelectItem>
                <SelectItem value="failed">Failed</SelectItem>
                <SelectItem value="skipped">Skipped</SelectItem>
              </SelectContent>
            </Select>

            <Input
              className="h-8 text-sm w-44"
              placeholder="Search recipient…"
              value={search}
              onChange={(e) => { setSearch(e.target.value); setOffset(0); }}
            />

            <div className="flex items-center gap-1">
              <Input
                type="date"
                className="h-8 text-sm w-36"
                value={from}
                onChange={(e) => { setFrom(e.target.value); setOffset(0); }}
              />
              <span className="text-muted-foreground text-xs">–</span>
              <Input
                type="date"
                className="h-8 text-sm w-36"
                value={to}
                onChange={(e) => { setTo(e.target.value); setOffset(0); }}
              />
            </div>

            <Button variant="ghost" size="sm" className="h-8 text-xs" onClick={handleReset}>
              Reset
            </Button>
          </div>

          {/* Table */}
          {isLoading ? (
            <div className="flex items-center justify-center h-32 text-muted-foreground text-sm">
              <Loader2 className="h-4 w-4 animate-spin mr-2" />Loading…
            </div>
          ) : logs.length === 0 ? (
            <div className="flex flex-col items-center justify-center h-32 text-muted-foreground text-sm gap-2">
              <Bell className="h-6 w-6 opacity-30" />
              <span>No notifications found</span>
            </div>
          ) : (
            <div className="rounded-md border">
              <Table>
                <TableHeader>
                  <TableRow>
                    <TableHead className="w-20">Channel</TableHead>
                    <TableHead>Recipient</TableHead>
                    <TableHead className="w-32">Trigger</TableHead>
                    <TableHead className="w-20">Status</TableHead>
                    <TableHead className="w-40">Sent At</TableHead>
                    <TableHead className="w-20"></TableHead>
                  </TableRow>
                </TableHeader>
                <TableBody>
                  {logs.map((log) => (
                    <TableRow key={log.id}>
                      <TableCell>
                        <div className="flex items-center gap-1.5 text-muted-foreground">
                          {channelIcon(log.channel)}
                          <span className="text-xs uppercase">{log.channel}</span>
                        </div>
                      </TableCell>
                      <TableCell className="font-mono text-xs">{log.recipient}</TableCell>
                      <TableCell className="text-xs text-muted-foreground">{log.trigger}</TableCell>
                      <TableCell>
                        <span className={`px-1.5 py-0.5 rounded text-[10px] font-medium ${statusClass(log.status)}`}>
                          {log.status}
                        </span>
                      </TableCell>
                      <TableCell className="text-xs text-muted-foreground">
                        {new Date(log.sentAt).toLocaleString()}
                      </TableCell>
                      <TableCell>
                        <Button
                          variant="ghost"
                          size="sm"
                          className="h-6 text-xs px-2"
                          onClick={() => setSelectedLog(log)}
                        >
                          Preview
                        </Button>
                      </TableCell>
                    </TableRow>
                  ))}
                </TableBody>
              </Table>
            </div>
          )}

          {/* Pagination */}
          {(offset > 0 || logs.length === LIMIT) && (
            <div className="flex items-center justify-between">
              <Button
                variant="outline"
                size="sm"
                className="h-8 gap-1"
                disabled={offset === 0}
                onClick={() => setOffset(Math.max(0, offset - LIMIT))}
              >
                <ChevronLeft className="h-3.5 w-3.5" />
                Previous
              </Button>
              <span className="text-xs text-muted-foreground">
                Showing {offset + 1}–{offset + logs.length}
              </span>
              <Button
                variant="outline"
                size="sm"
                className="h-8 gap-1"
                disabled={logs.length < LIMIT}
                onClick={() => setOffset(offset + LIMIT)}
              >
                Next
                <ChevronRight className="h-3.5 w-3.5" />
              </Button>
            </div>
          )}

          {/* Preview modal */}
          <Dialog open={!!selectedLog} onOpenChange={(open) => { if (!open) setSelectedLog(null); }}>
            <DialogContent className="max-w-lg">
              <DialogHeader>
                <DialogTitle className="flex items-center gap-2 text-sm">
                  {selectedLog && channelIcon(selectedLog.channel)}
                  <span>Notification Preview</span>
                  {selectedLog && (
                    <span className={`ml-auto px-1.5 py-0.5 rounded text-[10px] font-medium ${statusClass(selectedLog.status)}`}>
                      {selectedLog.status}
                    </span>
                  )}
                </DialogTitle>
              </DialogHeader>
              {selectedLog && (
                <div className="space-y-3 text-sm">
                  <div className="grid grid-cols-[5rem_1fr] gap-1 text-xs text-muted-foreground">
                    <span className="font-medium">Recipient</span>
                    <span className="font-mono">{selectedLog.recipient}</span>
                    <span className="font-medium">Trigger</span>
                    <span>{selectedLog.trigger}</span>
                    <span className="font-medium">Sent At</span>
                    <span>{new Date(selectedLog.sentAt).toLocaleString()}</span>
                    {selectedLog.providerMessageId && (
                      <>
                        <span className="font-medium">Provider ID</span>
                        <span className="font-mono break-all">{selectedLog.providerMessageId}</span>
                      </>
                    )}
                  </div>
                  <pre className="p-3 rounded bg-muted text-xs whitespace-pre-wrap break-all max-h-60 overflow-auto">
                    {selectedLog.preview}
                  </pre>
                  {selectedLog.errorMessage && (
                    <p className="text-xs text-red-600 dark:text-red-400">
                      Error: {selectedLog.errorMessage}
                    </p>
                  )}
                </div>
              )}
            </DialogContent>
          </Dialog>
        </div>
      );
    }
    ```
  </action>
