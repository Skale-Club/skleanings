<objective>
## Goal
Extend the availability engine with per-staff slot computation. When a specific staffId is passed to
the availability endpoints, slots are computed against that staff member's schedule and bookings
rather than the global business hours.

## Purpose
Customers selecting a specific staff member at booking time need availability filtered to that
professional. The existing global logic remains unchanged for deployments without staff.

## Output
- `server/storage.ts` — two new storage methods: getBookingsByDateAndStaff, getStaffMembersByServiceId
- `server/lib/staff-availability.ts` — new lib: getStaffAvailableSlots function
- `server/routes/availability.ts` — accept optional staffId query param, route to per-staff logic
</objective>

    import { storage } from "../storage";

    /**
     * Compute available time slots for a specific staff member on a date.
     * Uses the staff member's staffAvailability weekly schedule and their existing bookings.
     * Falls back to business hours if no availability record exists for that day.
     */
    export async function getStaffAvailableSlots(
      staffMemberId: number,
      date: string,
      durationMinutes: number,
      options?: { timeZone?: string }
    ): Promise<string[]> {
      const dayOfWeek = new Date(date + 'T12:00:00').getDay(); // 0=Sun, 6=Sat

      // Get this staff member's availability for this day of week
      const availability = await storage.getStaffAvailability(staffMemberId);
      const dayRecord = availability.find(a => a.dayOfWeek === dayOfWeek);

      if (!dayRecord || !dayRecord.isAvailable) return [];

      const [startHr, startMn] = dayRecord.startTime.split(':').map(Number);
      const [endHr, endMn] = dayRecord.endTime.split(':').map(Number);

      // Get bookings already assigned to this staff member on this date
      const existingBookings = await storage.getBookingsByDateAndStaff(date, staffMemberId);

      // Time-of-day awareness (skip past slots when date is today)
      const timeZone = options?.timeZone || 'America/New_York';
      const now = new Date();
      const tzNow = new Date(now.toLocaleString('en-US', { timeZone }));
      const todayStr = getTodayStr(tzNow);
      const isToday = date === todayStr;
      const currentHour = tzNow.getHours();
      const currentMinute = tzNow.getMinutes();

      const slots: string[] = [];

      for (let h = startHr; h < endHr || (h === endHr && 0 < endMn); h++) {
        for (let m = 0; m < 60; m += 30) {
          if (h === startHr && m < startMn) continue;
          if (h > endHr || (h === endHr && m >= endMn)) continue;

          const slotHour = h.toString().padStart(2, '0');
          const slotMinute = m.toString().padStart(2, '0');
          const startTime = `${slotHour}:${slotMinute}`;

          if (isToday) {
            if (h < currentHour || (h === currentHour && m <= currentMinute)) continue;
          }

          // Check slot end time fits within staff window
          const slotDate = new Date(`2000-01-01T${startTime}:00`);
          slotDate.setMinutes(slotDate.getMinutes() + durationMinutes);
          if (
            slotDate.getHours() > endHr ||
            (slotDate.getHours() === endHr && slotDate.getMinutes() > endMn)
          ) continue;

          const endHour = slotDate.getHours().toString().padStart(2, '0');
          const endMinute = slotDate.getMinutes().toString().padStart(2, '0');
          const endTime = `${endHour}:${endMinute}`;

          // Check no booking conflict for this staff member
          const hasConflict = existingBookings.some(
            b => startTime < b.endTime && endTime > b.startTime
          );
          if (!hasConflict) slots.push(startTime);
        }
      }

      return slots;
    }

    function getTodayStr(date: Date): string {
      return date.toISOString().split('T')[0];
    }
