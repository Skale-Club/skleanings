## Goal
Extend `UserDialog.tsx` with a Role selector dropdown (admin/staff/viewer) and a "Link to staff member" dropdown, so admins can manage user roles and staff account linkage from the Users section.

## Purpose
Completes the Phase 4 user management loop. Admin can promote a user to 'staff', then link them to a staff member — enabling staff login to access the scoped calendar view built in Plan 02.

  <action>
    Read the full UserDialog.tsx first to understand the form structure.

    Extend the dialog to add:

    1. Add to imports:
    ```ts
    import { Select, SelectContent, SelectItem, SelectTrigger, SelectValue } from '@/components/ui/select';
    import type { StaffMember } from '@shared/schema';
    ```

    2. Add state for role and staffLink (separate from the main form, since they hit different endpoints):
    ```ts
    const [role, setRole] = useState<'admin' | 'staff' | 'viewer'>(user?.role ?? 'viewer');
    const [staffLinkId, setStaffLinkId] = useState<number | null>(null);
    ```

    3. Fetch staff members for the link dropdown:
    ```ts
    const { data: staffMembers } = useQuery<StaffMember[]>({
      queryKey: ['/api/staff'],
      queryFn: async () => {
        const token = await getAccessToken();
        const res = await fetch('/api/staff?includeInactive=true', {
          headers: { Authorization: `Bearer ${token}` },
        });
        return res.json();
      },
    });
    ```

    4. Fetch current staff link for this user (to pre-populate dropdown):
    ```ts
    useEffect(() => {
      if (user && staffMembers) {
        const linked = staffMembers.find(s => s.userId === user.id);
        setStaffLinkId(linked?.id ?? null);
      }
    }, [user, staffMembers]);
    ```

    Also sync role when user prop changes:
    ```ts
    useEffect(() => {
      setRole(user?.role ?? 'viewer');
    }, [user]);
    ```

    5. Add role save + staff-link save mutations alongside the existing form submit:
    ```ts
    const saveRoleAndLink = async (userId: string) => {
      const token = await getAccessToken();
      if (!token) return;
      // Save role
      await fetch(`/api/users/${userId}`, {
        method: 'PATCH',
        headers: { 'Content-Type': 'application/json', Authorization: `Bearer ${token}` },
        body: JSON.stringify({ role }),
      });
      // Save staff link if role is staff
      if (role === 'staff') {
        await fetch(`/api/users/${userId}/staff-link`, {
          method: 'PATCH',
          headers: { 'Content-Type': 'application/json', Authorization: `Bearer ${token}` },
          body: JSON.stringify({ staffMemberId: staffLinkId }),
        });
      }
    };
    ```

    6. Call `saveRoleAndLink(userId)` inside the existing form submit `onSuccess` handler, after the user is created/updated.

    7. Add the UI fields BEFORE the form's submit button:
    ```tsx
    {/* Role selector */}
    <div className="space-y-1">
      <label className="text-sm font-medium">Role</label>
      <Select value={role} onValueChange={(v) => setRole(v as any)}>
        <SelectTrigger>
          <SelectValue />
        </SelectTrigger>
        <SelectContent>
          <SelectItem value="admin">Admin</SelectItem>
          <SelectItem value="staff">Staff</SelectItem>
          <SelectItem value="viewer">Viewer</SelectItem>
        </SelectContent>
      </Select>
    </div>

    {/* Staff link — only shown when role is staff */}
    {role === 'staff' && (
      <div className="space-y-1">
        <label className="text-sm font-medium">Link to Staff Member</label>
        <Select
          value={staffLinkId?.toString() ?? 'none'}
          onValueChange={(v) => setStaffLinkId(v === 'none' ? null : Number(v))}
        >
          <SelectTrigger>
            <SelectValue placeholder="Select staff member..." />
          </SelectTrigger>
          <SelectContent>
            <SelectItem value="none">— None —</SelectItem>
            {staffMembers?.map(s => (
              <SelectItem key={s.id} value={s.id.toString()}>
                {s.firstName} {s.lastName}
              </SelectItem>
            ))}
          </SelectContent>
        </Select>
      </div>
    )}
    ```
