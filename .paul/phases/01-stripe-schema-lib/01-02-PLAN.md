    1. **Platform key** (from env var `STRIPE_SECRET_KEY`) тАФ used ONLY for OAuth token exchange with Stripe Connect. This is the platform operator's key.

    2. **Connected account key** (from `integrationSettings.apiKey` = access_token) тАФ used for creating Checkout sessions on behalf of the connected account.

    Add a new exported function `getConnectAuthUrl()` that builds the Stripe Connect OAuth URL:
    ```typescript
    export function getConnectAuthUrl(): string {
      const clientId = process.env.STRIPE_CLIENT_ID;
      if (!clientId) throw new Error("STRIPE_CLIENT_ID env var not set.");
      const params = new URLSearchParams({
        response_type: "code",
        client_id: clientId,
        scope: "read_write",
        redirect_uri: process.env.STRIPE_REDIRECT_URI || "",
      });
      return `https://connect.stripe.com/oauth/authorize?${params.toString()}`;
    }
    ```

    Add `exchangeConnectCode(code: string)` that calls Stripe's token endpoint using the PLATFORM secret key:
    ```typescript
    export async function exchangeConnectCode(code: string): Promise<{ accessToken: string; stripeUserId: string }> {
      const secretKey = process.env.STRIPE_SECRET_KEY;
      if (!secretKey) throw new Error("STRIPE_SECRET_KEY env var not set.");
      const stripe = new Stripe(secretKey, { apiVersion: "2026-03-25.dahlia" });
      const response = await stripe.oauth.token({ grant_type: "authorization_code", code });
      if (!response.access_token || !response.stripe_user_id) {
        throw new Error("Stripe Connect token exchange failed");
      }
      return { accessToken: response.access_token, stripeUserId: response.stripe_user_id };
    }
    ```

    Add `deauthorizeConnectAccount(stripeUserId: string)` that revokes the connection:
    ```typescript
    export async function deauthorizeConnectAccount(stripeUserId: string): Promise<void> {
      const clientId = process.env.STRIPE_CLIENT_ID;
      const secretKey = process.env.STRIPE_SECRET_KEY;
      if (!clientId || !secretKey) return;
      const stripe = new Stripe(secretKey, { apiVersion: "2026-03-25.dahlia" });
      await stripe.oauth.deauthorize({ client_id: clientId, stripe_user_id: stripeUserId });
    }
    ```

    Keep all existing exports: `createCheckoutSession`, `retrieveCheckoutSession`, `verifyWebhookEvent`.

    Avoid: using the platform key for creating charges тАФ charges must go through the connected account's access_token.

      try {
        const settings = await storage.getIntegrationSettings("stripe");
        if (!settings?.apiKey) return res.json({ connected: false });
        res.json({

    **Add `PUT /stripe/webhook`** тАФ allows saving just the webhook secret (still needs manual entry from Stripe dashboard):
    ```typescript
    router.put("/stripe/webhook", requireAdmin, async (req, res) => {
      try {
        const { webhookSecret, isEnabled } = req.body;
        const existing = await storage.getIntegrationSettings("stripe");
        const toSave: any = {
          provider: "stripe",
          apiKey: existing?.apiKey ?? "",
          locationId: existing?.locationId ?? "",
          isEnabled: isEnabled ?? existing?.isEnabled ?? false,
        };
        if (webhookSecret && webhookSecret !== "********") {
          toSave.calendarId = webhookSecret;
        } else {
          toSave.calendarId = existing?.calendarId ?? "";
        }
        await storage.upsertIntegrationSettings(toSave);
        res.json({ success: true });
      } catch (err) {
        res.status(400).json({ message: (err as Error).message });
      }
    });
    ```

    **Add `GET /stripe/connect`** тАФ initiate OAuth (must be BEFORE `/:id` style routes):
    ```typescript
    router.get("/stripe/connect", requireAdmin, async (req, res) => {
      try {
        if (!process.env.STRIPE_CLIENT_ID) {
          return res.status(501).json({ message: "STRIPE_CLIENT_ID env var not configured on server." });
        }
        const url = getConnectAuthUrl();
        res.redirect(url);
      } catch (err) {
        res.status(500).json({ message: (err as Error).message });
      }
    });
    ```

    **Add `GET /stripe/callback`** тАФ exchange code, store tokens, redirect:
    ```typescript
    router.get("/stripe/callback", async (req, res) => {
      try {
        const code = String(req.query.code || "");
        if (!code) return res.status(400).send("Missing OAuth code");
        const { accessToken, stripeUserId } = await exchangeConnectCode(code);
        const existing = await storage.getIntegrationSettings("stripe");
        await storage.upsertIntegrationSettings({
          provider: "stripe",
          apiKey: accessToken,
          locationId: stripeUserId,
          calendarId: existing?.calendarId ?? "",
          isEnabled: existing?.isEnabled ?? true,
        });
        res.redirect("/admin/integrations");
      } catch (err) {
        res.status(500).send(`Stripe Connect error: ${(err as Error).message}`);
      }
    });
    ```

    **Add `DELETE /stripe/disconnect`** тАФ revoke and clear:
    ```typescript
    router.delete("/stripe/disconnect", requireAdmin, async (req, res) => {
      try {
        const existing = await storage.getIntegrationSettings("stripe");
        if (existing?.locationId) {
          await deauthorizeConnectAccount(existing.locationId);
        }
        await storage.upsertIntegrationSettings({
          provider: "stripe",
          apiKey: "",
          locationId: "",
          calendarId: existing?.calendarId ?? "",
          isEnabled: false,
        });
        res.json({ success: true });
      } catch (err) {
        res.status(500).json({ message: (err as Error).message });
      }
    });
    ```
