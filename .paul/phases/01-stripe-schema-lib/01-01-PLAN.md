    Create server/lib/stripe.ts:

    ```typescript
    import Stripe from "stripe";
    import { storage } from "../storage";

    async function getStripeClient(): Promise<Stripe> {
      const creds = await storage.getIntegrationSettings("stripe");
      if (!creds?.apiKey) {
        throw new Error("Stripe not configured. Add secret key in Admin → Integrations.");
      }
      return new Stripe(creds.apiKey, { apiVersion: "2024-12-18.acacia" });
    }

    export interface CheckoutSessionParams {
      bookingId: number;
      amountCents: number;        // total price in cents (integer)
      customerEmail?: string;
      customerName?: string;
      lineItems: { name: string; amount: number; quantity: number }[];
      successUrl: string;         // must include {CHECKOUT_SESSION_ID} placeholder
      cancelUrl: string;
    }

    export async function createCheckoutSession(params: CheckoutSessionParams): Promise<Stripe.Checkout.Session> {
      const stripe = await getStripeClient();
      return stripe.checkout.sessions.create({
        payment_method_types: ["card"],
        mode: "payment",
        customer_email: params.customerEmail,
        line_items: params.lineItems.map(item => ({
          price_data: {
            currency: "usd",
            product_data: { name: item.name },
            unit_amount: item.amount,   // already in cents
          },
          quantity: item.quantity,
        })),
        metadata: { bookingId: String(params.bookingId) },
        success_url: params.successUrl,
        cancel_url: params.cancelUrl,
      });
    }

    export async function retrieveCheckoutSession(sessionId: string): Promise<Stripe.Checkout.Session> {
      const stripe = await getStripeClient();
      return stripe.checkout.sessions.retrieve(sessionId);
    }

    export async function verifyWebhookEvent(rawBody: Buffer, signature: string): Promise<Stripe.Event> {
      const creds = await storage.getIntegrationSettings("stripe");
      if (!creds?.calendarId) {
        throw new Error("Stripe webhook secret not configured.");
      }
      const stripe = await getStripeClient();
      return stripe.webhooks.constructEvent(rawBody, signature, creds.calendarId);
    }
    ```

    Use the integrationSettings field mapping:
    - `apiKey` = Stripe Secret Key (sk_live_... or sk_test_...)
    - `locationId` = Stripe Publishable Key (pk_live_... or pk_test_...)
    - `calendarId` = Stripe Webhook Secret (whsec_...)
    - `isEnabled` = integration on/off toggle

    Avoid: hardcoding API version — use the latest stable one listed above.

    ```typescript
    // ===============================
    // Stripe Integration Routes
    // ===============================

    router.get("/stripe", requireAdmin, async (req, res) => {
      try {
        const settings = await storage.getIntegrationSettings("stripe");
        if (!settings) return res.json({ provider: "stripe", apiKey: "", locationId: "", calendarId: "", isEnabled: false });
        res.json({
          ...settings,
          apiKey: settings.apiKey ? "********" : "",
          locationId: settings.locationId ? "********" : "",
          calendarId: settings.calendarId ? "********" : "",
        });
      } catch (err) {
        res.status(500).json({ message: (err as Error).message });
      }
    });

    router.put("/stripe", requireAdmin, async (req, res) => {
      try {
        const { apiKey, locationId, calendarId, isEnabled } = req.body;
        const existing = await storage.getIntegrationSettings("stripe");
        const toSave: any = { provider: "stripe", isEnabled: isEnabled ?? false };
        toSave.apiKey = (apiKey && apiKey !== "********") ? apiKey : (existing?.apiKey ?? "");
        toSave.locationId = (locationId && locationId !== "********") ? locationId : (existing?.locationId ?? "");
        toSave.calendarId = (calendarId && calendarId !== "********") ? calendarId : (existing?.calendarId ?? "");
        const saved = await storage.upsertIntegrationSettings(toSave);
        res.json({
          ...saved,
          apiKey: saved.apiKey ? "********" : "",
          locationId: saved.locationId ? "********" : "",
          calendarId: saved.calendarId ? "********" : "",
        });
      } catch (err) {
        res.status(400).json({ message: (err as Error).message });
      }
    });
    ```
