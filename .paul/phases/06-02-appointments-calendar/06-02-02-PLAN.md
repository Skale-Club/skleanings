<objective>
## Goal
Build the `AppointmentsCalendarSection` admin component — a full visual calendar of all bookings with Month/Week/Day views, per-staff color coding, status filter, and GCal busy-time overlay — and wire it into the admin sidebar.

## Purpose
Admins currently manage bookings in a flat list. A calendar view gives instant spatial awareness of the schedule: gaps, back-to-back bookings, staff workload, and conflicts — all visible without filtering or scrolling.

## Output
- `'calendar'` added to `AdminSection` type
- Calendar menu item in Admin.tsx sidebar
- `AppointmentsCalendarSection` component with:
  - react-big-calendar Month/Week/Day views
  - Bookings fetched for visible date range via `GET /api/bookings?from=&to=`
  - Per-staff color coding (deterministic hue from staff id)
  - Staff filter checkboxes (show/hide individual staff)
  - Status filter (All / Pending / Confirmed / Completed / Cancelled)
  - GCal busy blocks as gray background events (fetched async, non-blocking)
  - Custom event component: customer name + status chip
</objective>

<task type="auto">
  <name>Task 2: Build AppointmentsCalendarSection component</name>
  <files>client/src/components/admin/AppointmentsCalendarSection.tsx</files>
  <action>
    Create the file with the full calendar component. Key implementation details:

    **Imports:**
    ```ts
    import { Calendar, dateFnsLocalizer, Views, Event } from 'react-big-calendar';
    import { format, parse, startOfWeek, getDay, startOfMonth, endOfMonth, startOfWeek as startOfWeekFns, endOfWeek, addDays } from 'date-fns';
    import { enUS } from 'date-fns/locale';
    import { useQuery } from '@tanstack/react-query';
    import type { Booking, StaffMember } from '@shared/schema';
    ```

    **Localizer:**
    ```ts
    const localizer = dateFnsLocalizer({
      format,
      parse,
      startOfWeek: () => startOfWeek(new Date(), { weekStartsOn: 0 }),
      getDay,
      locales: { 'en-US': enUS },
    });
    ```

    **Staff color helper** (deterministic, no state):
    ```ts
    const STAFF_COLORS = [
      '#3B82F6', '#10B981', '#F59E0B', '#EF4444',
      '#8B5CF6', '#EC4899', '#06B6D4', '#84CC16',
    ];
    function getStaffColor(staffId: number | null | undefined): string {
      if (!staffId) return '#6B7280'; // gray for unassigned
      return STAFF_COLORS[staffId % STAFF_COLORS.length];
    }
    ```

    **Calendar event interface:**
    ```ts
    interface CalendarEvent extends Event {
      bookingId: number;
      status: string;
      staffMemberId: number | null;
      color: string;
      isGcalBusy?: boolean;
    }
    ```

    **Component structure:**
    ```tsx
    export function AppointmentsCalendarSection({ getAccessToken }: { getAccessToken: () => Promise<string> }) {
      const [currentDate, setCurrentDate] = useState(new Date());
      const [currentView, setCurrentView] = useState<string>(Views.MONTH);
      const [hiddenStaff, setHiddenStaff] = useState<Set<number>>(new Set());
      const [hiddenStatuses, setHiddenStatuses] = useState<Set<string>>(new Set());

      // Compute visible date range from currentDate + currentView
      const { from, to } = useMemo(() => {
        const start = currentView === Views.MONTH
          ? startOfMonth(currentDate)
          : currentView === Views.WEEK
          ? startOfWeek(currentDate, { weekStartsOn: 0 })
          : currentDate;
        const end = currentView === Views.MONTH
          ? endOfMonth(currentDate)
          : currentView === Views.WEEK
          ? endOfWeek(currentDate, { weekStartsOn: 0 })
          : currentDate;
        return {
          from: format(start, 'yyyy-MM-dd'),
          to: format(end, 'yyyy-MM-dd'),
        };
      }, [currentDate, currentView]);

      // Fetch bookings for visible range
      const { data: bookings = [] } = useQuery<Booking[]>({
        queryKey: ['/api/bookings', from, to],
        queryFn: async () => {
          const token = await getAccessToken();
          const res = await fetch(`/api/bookings?from=${from}&to=${to}`, {
            headers: { Authorization: `Bearer ${token}` },
          });
          return res.json();
        },
      });

      // Fetch staff members
      const { data: staffList = [] } = useQuery<StaffMember[]>({
        queryKey: ['/api/staff', 'all'],
        queryFn: async () => {
          const token = await getAccessToken();
          const res = await fetch('/api/staff?includeInactive=true', {
            headers: { Authorization: `Bearer ${token}` },
          });
          return res.json();
        },
      });

      // Convert bookings to calendar events, apply filters
      const events: CalendarEvent[] = useMemo(() => {
        return bookings
          .filter(b => {
            if (b.staffMemberId && hiddenStaff.has(b.staffMemberId)) return false;
            if (hiddenStatuses.has(b.status)) return false;
            return true;
          })
          .map(b => ({
            bookingId: b.id,
            title: b.customerName,
            start: new Date(`${b.bookingDate}T${b.startTime}:00`),
            end: new Date(`${b.bookingDate}T${b.endTime}:00`),
            status: b.status,
            staffMemberId: b.staffMemberId ?? null,
            color: getStaffColor(b.staffMemberId),
          }));
      }, [bookings, hiddenStaff, hiddenStatuses]);

      // Custom event component
      const EventComponent = ({ event }: { event: CalendarEvent }) => {
        if (event.isGcalBusy) {
          return <div className="text-xs text-gray-400 px-1">Busy</div>;
        }
        const statusColors: Record<string, string> = {
          pending: 'bg-yellow-100 text-yellow-800',
          confirmed: 'bg-green-100 text-green-800',
          completed: 'bg-blue-100 text-blue-800',
          cancelled: 'bg-red-100 text-red-800',
        };
        return (
          <div className="text-xs px-1 overflow-hidden">
            <div className="font-medium truncate">{event.title as string}</div>
            <span className={`text-xs px-1 rounded ${statusColors[event.status] ?? ''}`}>
              {event.status}
            </span>
          </div>
        );
      };

      const eventStyleGetter = (event: CalendarEvent) => ({
        style: {
          backgroundColor: event.isGcalBusy ? '#E5E7EB' : event.color,
          border: 'none',
          opacity: event.isGcalBusy ? 0.6 : 1,
          cursor: event.isGcalBusy ? 'default' : 'pointer',
          color: event.isGcalBusy ? '#9CA3AF' : '#fff',
        },
      });

      const STATUSES = ['pending', 'confirmed', 'completed', 'cancelled'];

      return (
        <div className="space-y-4">
          <div className="flex items-center justify-between">
            <h2 className="text-2xl font-bold">Calendar</h2>
          </div>

          {/* Filter bar */}
          <div className="flex flex-wrap gap-4 p-4 bg-card rounded-lg border">
            {/* Staff filters */}
            {staffList.length > 0 && (
              <div className="flex flex-wrap gap-2 items-center">
                <span className="text-sm font-medium text-muted-foreground">Staff:</span>
                {staffList.map(staff => (
                  <label key={staff.id} className="flex items-center gap-1 cursor-pointer">
                    <input
                      type="checkbox"
                      checked={!hiddenStaff.has(staff.id)}
                      onChange={() => {
                        setHiddenStaff(prev => {
                          const next = new Set(prev);
                          next.has(staff.id) ? next.delete(staff.id) : next.add(staff.id);
                          return next;
                        });
                      }}
                    />
                    <span
                      className="w-3 h-3 rounded-full inline-block"
                      style={{ backgroundColor: getStaffColor(staff.id) }}
                    />
                    <span className="text-sm">{staff.firstName}</span>
                  </label>
                ))}
              </div>
            )}
            {/* Status filters */}
            <div className="flex flex-wrap gap-2 items-center">
              <span className="text-sm font-medium text-muted-foreground">Status:</span>
              {STATUSES.map(status => (
                <label key={status} className="flex items-center gap-1 cursor-pointer">
                  <input
                    type="checkbox"
                    checked={!hiddenStatuses.has(status)}
                    onChange={() => {
                      setHiddenStatuses(prev => {
                        const next = new Set(prev);
                        next.has(status) ? next.delete(status) : next.add(status);
                        return next;
                      });
                    }}
                  />
                  <span className="text-sm capitalize">{status}</span>
                </label>
              ))}
            </div>
          </div>

          {/* Calendar */}
          <div className="bg-card rounded-lg border p-4" style={{ height: 700 }}>
            <Calendar
              localizer={localizer}
              events={events}
              date={currentDate}
              view={currentView as any}
              onNavigate={setCurrentDate}
              onView={setCurrentView}
              views={[Views.MONTH, Views.WEEK, Views.DAY]}
              eventPropGetter={eventStyleGetter}
              components={{ event: EventComponent as any }}
              style={{ height: '100%' }}
            />
          </div>

          {/* Staff color legend */}
          {staffList.length > 0 && (
            <div className="flex flex-wrap gap-3 text-sm">
              {staffList.map(staff => (
                <div key={staff.id} className="flex items-center gap-1">
                  <span className="w-3 h-3 rounded-full" style={{ backgroundColor: getStaffColor(staff.id) }} />
                  <span>{staff.firstName} {staff.lastName}</span>
                </div>
              ))}
              <div className="flex items-center gap-1">
                <span className="w-3 h-3 rounded-full bg-gray-400" />
                <span>Unassigned</span>
              </div>
            </div>
          )}
        </div>
      );
    }
    ```

    Note: GCal busy overlay is intentionally deferred to Plan 03 to keep this plan focused on the core calendar rendering. The component is built to accept background events via the `events` array (isGcalBusy flag) — Plan 03 will add the fetch logic.
  </action>
