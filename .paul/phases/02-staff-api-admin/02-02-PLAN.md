    Avatar component (inline, no external dep):
    ```tsx
    function StaffAvatar({ member }: { member: StaffMember }) {
      if (member.profileImageUrl) {
        return <img src={member.profileImageUrl} alt="" className="w-9 h-9 rounded-full object-cover shrink-0" />;
      }
      const initials = `${member.firstName[0]}${member.lastName[0]}`.toUpperCase();
      return (
        <div className="w-9 h-9 rounded-full bg-primary/10 text-primary flex items-center justify-center text-sm font-semibold shrink-0">
          {initials}
        </div>
      );
    }
    ```

    Staff form fields (StaffForm component):
    - firstName: Input required
    - lastName: Input required
    - email: Input type="email"
    - phone: Input type="tel"
    - bio: Textarea rows={3}
    - profileImageUrl: Input for URL + small file upload button (uploads via uploadFileToServer, sets the URL field on success)
    - isActive: Switch with label "Active"

    Mutations:
    - createStaff: POST /api/staff → 201
    - updateStaff: PUT /api/staff/:id
    - deleteStaff: DELETE /api/staff/:id
    - toggleStaff: PUT /api/staff/:id with { isActive }
    - reorderStaff: PUT /api/staff/reorder with { updates: [{ id, order }] }
